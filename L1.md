#define _GNU_SOURCE
#include <stdio.h>
#include <stdint.h>
#include <stdlib.h>
#include <pthread.h>
#include <sched.h>
#include <time.h>

#define CORES 4
#define ITERS 800000

void* run(void* arg){
    int id = *(int*)arg;

    cpu_set_t set;
    CPU_ZERO(&set);
    CPU_SET(id, &set);
    sched_setaffinity(0, sizeof(set), &set);

    uint32_t v20 = 0x3D4CCCCD;
    uint32_t phi = 0x3FCF1BBC;
    uint32_t loops = ITERS;

    asm volatile(
        "vdup.32 q0, %[v20]        \n"
        "vdup.32 q1, %[phi]        \n"
        "vdup.32 q2, %[v20]        \n"
        "vdup.32 q3, %[phi]        \n"

        "1:                        \n"
        "vmla.f32 q0, q1, q2       \n"
        "vmla.f32 q3, q0, q1       \n"
        "vmls.f32 q1, q3, q2       \n"
        "vmls.f32 q2, q0, q3       \n"

        "vext.32 q4, q0, q1, #1    \n"
        "vrev64.32 q5, q3          \n"
        "veor q0, q4, q5           \n"

        "subs %[cnt], %[cnt], #1   \n"
        "bne 1b                    \n"

        : [cnt] "+r"(loops)
        : [v20]"r"(v20), [phi]"r"(phi)
        : "q0","q1","q2","q3","q4","q5","cc"
    );

    return NULL;
}

int main(){
    pthread_t t[CORES];
    int id[CORES];
    struct timespec s,e;

    clock_gettime(CLOCK_MONOTONIC, &s);

    for(int i=0;i<CORES;i++){
        id[i]=i;
        pthread_create(&t[i],0,run,&id[i]);
    }
    for(int i=0;i<CORES;i++) pthread_join(t[i],0);

    clock_gettime(CLOCK_MONOTONIC, &e);

    double dt = (e.tv_sec - s.tv_sec) + (e.tv_nsec - s.tv_nsec)/1e9;

    printf("\n[KERNEL EXTREMO ARM32]\nTempo: %.4f s\n", dt);

    return 0;
}
EOF
